    {
    \begin{center}
    \begin{tabular}{p{0.9\textwidth}}
    \cellcolor{#2}
    \textbf{#1}: \\
    }
    {
    \\\\
    \end{tabular}
    \end{center}
    }

after title={: \\}, coltitle=black,colback=gray!10,colframe=black, title={#2},
fonttitle=\bfseries,##1}}
